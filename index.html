async function fetchSheetData() { const loadingUI = document.getElementById('loadingUI'); const refreshIcon = document.getElementById('refreshIcon'); loadingUI.classList.remove('hidden'); refreshIcon.classList.add('fa-spin'); try { const response = await fetch(SHEET_URL); const csvData = await response.text(); // 엑셀/구글시트의 CSV 특성을 고려한 행 분리 (따옴표 안의 줄바꿈 보호) const rows = csvData.match(/(".*?"|[^"\r\n]+)(?=\r*\n|$)|(?<=\n|^)\r*\n/g) || []; programs = rows.slice(1).map(row => { // 쉼표로 분리하되 따옴표로 묶인 내용은 하나로 취급 const cols = []; let current = ''; let inQuotes = false; for (let i = 0; i < row.length; i++) { const char = row[i]; if (char === '"') inQuotes = !inQuotes; else if (char === ',' && !inQuotes) { cols.push(current.trim()); current = ''; } else { current += char; } } cols.push(current.trim()); // 양 끝 따옴표 제거 및 내부 쌍따옴표 정리 const clean = cols.map(c => c.replace(/^"|"$/g, '').replace(/""/g, '"').trim()); return { title: clean[0] || '', // A열: 프로그램명 status: clean[1] || '접수중', // B열: 접수유무 desc: clean[2] || '', // C열: 설명 submission: clean[3] || '', // D열: 접수방법 url: clean[4] || '#' // E열: 신청링크 or 제출처 }; }).filter(p => p.title && p.title !== ""); renderContent(); } catch (error) { console.error(error); document.getElementById('errorUI').classList.remove('hidden'); } finally { loadingUI.classList.add('hidden'); refreshIcon.classList.remove('fa-spin'); } }
